#include <Arduino.h>
#include <WiFi.h>
#include <WiFiManager.h>

WiFiManager wm;
// put function declarations here:
int myFunction(int, int);
void startPortal(); // Function prototype




void setup() {
  // put your setup code here, to run once:
  int result = myFunction(2, 3);
  Serial.begin(115200);
  Serial.println("ESP32 is alive!");

  bool hasCreds = WiFi.SSID().length() > 0;

  if (!hasCreds) {
    Serial.println("No WiFi credentials found, starting portal");
    startPortal();
    return;
  }

  Serial.println("WiFi credentials found, connecting...");
  if (!wm.autoConnect()) {
    Serial.println("Failed to connect, starting portal");
    startPortal();
  }

  Serial.println("Connected!");
  Serial.println(WiFi.localIP());

  pinMode(0, INPUT_PULLUP);
  if (digitalRead(0) == LOW) {
    wm.resetSettings();
    startPortal();
  }

}


void startPortal() {
  wm.setConfigPortalBlocking(true);
  wm.setConfigPortalTimeout(0); // no timeout

  wm.startConfigPortal("ESP32-Setup", "12345678");

  Serial.println("WiFi configured, rebooting...");
  delay(1000);
  ESP.restart();
}


void loop() {
  // put your main code here, to run repeatedly:
  delay(1000);
  Serial.println("Tick");
}
